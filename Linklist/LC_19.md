public ListNode removeNthFromEnd(ListNode head, int n) {
    ListNode ead = new ListNode();
    ead.next = head;
    ListNode slow = ead;
    ListNode fast = ead;
    for(int i = 0; i<n; i++){
        fast = fast.next;
    }
    while(fast.next!=null){
        slow = slow.next;
        fast = fast.next;
    }
    slow.next = slow.next.next;
    return ead.next;

}
